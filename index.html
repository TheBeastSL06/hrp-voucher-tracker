🔒 Firestore Security Rules

Copy and paste in Firebase Console → Firestore Database → Rules, then click Publish.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAuth()  { return request.auth != null; }
    function role()    { return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role; }
    function isAdmin() { return isAuth() && role() == 'superadmin'; }

    match /vouchers/{id} {
      allow read:           if true;
      allow create, delete: if isAdmin();
      allow update:         if isAuth();
    }
    match /dynamicSubCats/{id}    { allow read: if true; allow write: if isAdmin(); }
    match /dynamicSubSubCats/{id} { allow read: if true; allow write: if isAdmin(); }
    match /users/{uid} {
      allow read:   if isAuth() && (request.auth.uid == uid || isAdmin());
      allow create: if isAuth() && request.auth.uid == uid;
      allow update: if isAuth() && (request.auth.uid == uid || isAdmin());
    }
  }
}
